1. SELECT c.login, COUNT(ord.”inDelivery”) 
FROM "Couriers" AS c 
INNER JOIN "Orders" AS ord ON c.id = ord."courierId" 
WHERE ord.”inDelivery” = true GROUP BY c.login ord.”inDelivery”;



2. SELECT
  "Orders".track,
  CASE
    WHEN "Orders".finished = true THEN 2
    WHEN "Orders".cancelled = true THEN -1
    WHEN "Orders"."inDelivery" = true THEN 1
    ELSE 0
  END AS status
FROM
  "Orders";
